## Technical Requirements

### Database Schema (Supabase)
```sql
-- Users table
create table shiftcard_users (
  id uuid primary key default gen_random_uuid(),
  email text unique not null,
  password_hash text,
  tier text check (tier in ('basic', 'pro', 'custom')),
  stripe_customer_id text,
  created_at timestamp default now()
);

-- Cards table
create table shiftcards (
  id uuid primary key default gen_random_uuid(),
  user_id uuid references shiftcard_users(id),
  slug text unique not null,
  tier text not null,
  theme text default 'black',

  -- Contact info
  name text not null,
  title text,
  company text,
  email text,
  phone text,
  website text,
  tagline text,
  bio text,

  -- Avatar
  avatar_type text check (avatar_type in ('initials', 'photo')),
  avatar_url text,

  -- Socials (JSONB for flexibility)
  socials jsonb default '{}',

  -- Custom options (Pro/Custom)
  custom_domain text,
  accent_color text,
  show_branding boolean default true,
  animations_enabled boolean default false,
  custom_css text,

  -- Metadata
  published boolean default true,
  created_at timestamp default now(),
  updated_at timestamp default now()
);

-- Analytics table (Pro/Custom only)
create table shiftcard_analytics (
  id uuid primary key default gen_random_uuid(),
  card_id uuid references shiftcards(id),
  event_type text check (event_type in ('view', 'save', 'click')),
  click_target text, -- which link was clicked
  referrer text,
  user_agent text,
  ip_hash text, -- hashed for privacy
  created_at timestamp default now()
);

-- Indexes
create index idx_shiftcards_slug on shiftcards(slug);
create index idx_shiftcards_user on shiftcards(user_id);
create index idx_analytics_card on shiftcard_analytics(card_id);
```
